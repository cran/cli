# transform [plain]

    Code
      local({
        cli_text("This is a {.field field} (before)")

        cli_div(theme = list(span.field = list(transform = foo)))

        cli_end()
        cli_text("This is a {.field field} (after)")
      })
    Message
      This is a field (before)
      This is a FIELD (during)
      This is a field (after)
